Creating and Editing Tables


CREATE Table TallestTowers {
  id INT,
  state VARCHAR(200),
  city VARCHAR(200),
  name VARCHAR(200),
  height INT,
  floors INT,
  year_built INT
}

NSERT INTO TallestTower (id, state, city, name, height, floors, year_built)
VALUES (1, 'New York', 'New York City', 'One World Trade Center', '541', '104', '2013')

Retrieve Data

Basics


SELECT city, name, height FROM TallestTower
SELECT city, name, height FROM TallestTower ORDER BY height DESC
SELECT city, name, height FROM TallestTower WHERE height > 100

More Compelex


SELECT DISTINCT city FROM TallestTower
=>New York City, Chicago, Atlanta, Los Angeles, Philadelphia

SELECT city, name, height FROM TallestTower WHERE height > 100 OR state = 'Georgia'
